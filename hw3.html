numbers_df <- data.frame(word=as.character(0:50))

results <- data |> unnest_tokens(output=word, input=abstract)|>
  anti_join(stop_words, by = c("word")) |>
  anti_join(numbers_df, by = c("word")) |>
  count(word, term) |>
  bind_tf_idf(term=word, document=term, n=n) |>
  group_by(term) |>
  top_n(n=5, wt=tf_idf) |>
  arrange(desc(tf_idf)) |>
  ungroup() |>
  arrange(term)

print(n=50, results)
